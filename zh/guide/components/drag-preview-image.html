DragPreviewImage #

将 HTML Image 元素呈现为拖动预览的组件。

用法 #

<script lang="ts" setup>
import { DragPreviewImage, useDrag } from 'vue3-dnd'
const [collect, drag, dragPreview] = useDrag(() => ({
  type: 'Box',
}))
</script>

<template>
    <DragPreviewImage :connect="dragPreview" src="house_dragged.png" />
    <div :ref="drag">🏠</div>
</template>
